266 AeroMessages AeroModule_Web::sendCommand(void *CommandToSend)
267 {
268  AeroMessages SequenceIDToSend = ((AeroCommonTemplate *)CommandToSend)->SequenceID;
269  AeroMessages ReceivedSequenceID = NULL;
270  if (Debug)
271  {
272  logToSerials(F("Sending SequenceID:"), false, 3);
273  logToSerials(SequenceIDToSend, false, 1);
274  logToSerials(F("-"), false, 1);
275  logToSerials(toText_aeroSequenceID(SequenceIDToSend), false, 1);
276  logToSerials(F("and waiting for Acknowledgment..."), true, 1);
277  }
278  Parent->Wireless->openWritingPipe(WirelessChannel);
279  Parent->Wireless->flush_rx();
280  if (Parent->Wireless->write(CommandToSend, WirelessPayloadSize))
281  {
282  delay(50); //< give a little time to the nRF024L01+ chip to update the isAckPayloadAvailable flag
283  if (Parent->Wireless->isAckPayloadAvailable())
284  {
285  OnlineStatus = true;
286  Parent->Wireless->read(ReceivedResponse, WirelessPayloadSize);
287  ReceivedSequenceID = ((AeroCommonTemplate *)ReceivedResponse)->SequenceID;
288  if (*Debug)
289  {
290  logToSerials(F("Response SequenceID:"), false, 4);
291  logToSerials(ReceivedSequenceID, false, 1);
292  logToSerials(F("-"), false, 1);
293  logToSerials(toText_aeroSequenceID(ReceivedSequenceID), true, 1);
294  }
295 
296  switch (ReceivedSequenceID)
297  {
298  case AeroMessages::AeroModuleResponse1:
299  memcpy(&AeroModuleResponse1Received, ReceivedResponse, sizeof(struct AeroModuleResponse));
300  if (*Debug)
301  {
302  logToSerials(F("Module1:"), false, 4);
303  logToSerials(AeroModuleResponse1Received.Status, true, 1);
304  }
305  break;
306  case AeroMessages::AeroResponse1:
307  memcpy(&AeroResponse1Received, ReceivedResponse, sizeof(struct AeroResponse));
308  if (AeroCommand1ToSend.SprayEnabled || AeroCommand1ToSend.SprayDisabled || AeroCommand1ToSend.SprayNow || AeroCommand1ToSend.SprayOff || AeroCommand1ToSend.PumpOn || AeroCommand1ToSend.PumpOff || AeroCommand1ToSend.PumpDisable || AeroCommand1ToSend.MixReservoir || AeroCommand1ToSend.RefillPressureTank)
309  {
310  SyncRequested = true;
311  AeroCommand1ToSend.SprayEnabled = false;
312  AeroCommand1ToSend.SprayDisabled = false;
313  AeroCommand1ToSend.SprayNow = false;
314  AeroCommand1ToSend.SprayOff = false;
315  AeroCommand1ToSend.PumpOn = false;
316  AeroCommand1ToSend.PumpOff = false;
317  AeroCommand1ToSend.PumpDisable = false;
318  AeroCommand1ToSend.MixReservoir = false;
319  AeroCommand1ToSend.RefillPressureTank = false;
320  }
321  if (*Debug)
322  {
323  logToSerials(F("Aero1:"), false, 4);
324  logToSerials(AeroResponse1Received.PressureTankPresent, false, 1);
325  logToSerials(F(","), false, 1);
326  logToSerials(AeroResponse1Received.SprayEnabled, false, 1);
327  logToSerials(F(","), false, 1);
328  logToSerials(AeroResponse1Received.State, false, 1);
329  logToSerials(F(","), false, 1);
330  logToSerials(AeroResponse1Received.Pressure, false, 1);
331  logToSerials(F(","), false, 1);
332  logToSerials(AeroResponse1Received.LastSprayPressure, true, 1);
333  }
334  break;
335  case AeroMessages::AeroGetNext:
336  if (*Debug)
337  {
338  logToSerials(F("Last message received"), true, 4);
339  }
340  break;
341  default:
342  logToSerials(F("SequenceID not known, ignoring package"), true, 4);
343  break;
344  }
345  LastResponseReceived = millis();
346  }
347  else
348  {
349  if (*Debug)
350  logToSerials(F("Acknowledgement received without any data"), true, 4);
351  OnlineStatus = false;
352  }
353  }
354  else
355  {
356  if (*Debug)
357  logToSerials(F("No response"), true, 3);
358  OnlineStatus = false;
359  }
360  return ReceivedSequenceID;
361 }
